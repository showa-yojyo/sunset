```glsl
float doublePolynomialSigmoid(float x, float n){
    float left = pow(2. * x, n) / 2.;
    float right = pow(2. * (x - 1.), n) / 2.;
    float stp = step(0.5, x);
    return mix(
        // even polynomial
        mix(left, 1. - right, stp),
        // odd polynomial
        mix(left, 1. + right, stp),
        mod(n, 2.));
}
```
