図書館逃避行の `Makefile` に `docker build` などのための処方箋を追加。

```Makefile
DOCKER_FILE ?= Dockerfile
DOCKER_REPO ?= $(prefix)
DOCKER_TARGET ?= $(DOCKER_REGISTRY)/$(DOCKER_REPO):$(DOCKER_TAG)

.PHONY: docker-build
docker-build: $(source-videos)
	@docker build -f $(DOCKER_FILE) -t $(DOCKER_TARGET) .

.PHONY: docker-clean
docker-clean:
	@docker rmi $(DOCKER_TARGET) || true

.PHONY: docker-push
docker-push:
	@docker push $(DOCKER_TARGET)

.PHONY: docker-verify
docker-verify:
	@docker create $(DOCKER_TARGET) dummy &&
	docker export `docker ps -lq` | tar tf - | grep -- src &&
	docker rm `docker ps -lq`
```
